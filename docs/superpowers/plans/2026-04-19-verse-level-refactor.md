# Verse-Level Refactor Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the page-number arithmetic walker (`advanceByPages`) with a verse-by-verse weight accumulator (`walkByWeight`) that matches the Python `generate_plan_by_amount` reference exactly, across all three algorithms.

**Architecture:** A build-time codegen script reads `verses.csv` and `surahs.csv` and emits `frontend/src/lib/quran/verseData.ts` — a committed TypeScript file with typed arrays and lookup maps. All three algorithm files (`newMemorization.ts`, `minorRevision.ts`, `majorRevision.ts`) become consumers of the new `walkByWeight` function in `helpers.ts`. The waterfall model logic in `majorRevision.ts` is preserved unchanged; only the measurement engine (page-number arithmetic → weight accumulation) changes.

**Tech Stack:** Node.js + TypeScript (codegen script), Vitest (tests). No new npm packages needed — `fs` + `path` are built-in.

```typescript
// scripts/generate-verse-data.ts
// Usage: npx tsx scripts/generate-verse-data.ts
import * as fs from "fs";
import * as path from "path";

const ROOT = path.join(__dirname, "..");
const OUT = path.join(ROOT, "frontend", "src", "lib", "quran", "verseData.ts");

function parseCSV(filePath: string): Record<string, string>[] {
  const raw = fs.readFileSync(filePath, "utf-8").replace(/^\uFEFF/, "");
  const lines = raw.split(/\r?\n/).filter(Boolean);
  const headers = lines[0].split(",").map((h) => h.trim());
  return lines.slice(1).map((line) => {
    const parts = line.split(",");
    const record: Record<string, string> = {};
    headers.forEach((h, i) => {
      record[h] = (parts[i] ?? "").trim().replace(/^"|"$/g, "");
    });
    return record;
  });
}

const verseRows = parseCSV(path.join(ROOT, "verses.csv"));
const surahRows = parseCSV(path.join(ROOT, "surahs.csv"));

const verseLines = verseRows.map((r) => {
  const beginVerse = r["begin_verse"].replace(/\\/g, "\\\\").replace(/"/g, '\\"');
  return (
    `  { surahId:${r["surah_id"]}, ayah:${r["order_in_surah"]}, ` +
    `orderInQuran:${r["order_in_quraan"]}, reverseIndex:${r["reverse_index"]}, ` +
    `pageNo:${r["page_no"]}, lettersCount:${r["letters_count"]}, ` +
    `weightOnPage:${r["weight_on_page"]}, beginVerse:"${beginVerse}" }`
  );
});

const surahLines = surahRows.map((r) => {
  const name = r["name"].replace(/"/g, '\\"');
  return `  { id:${r["surah_id"]}, name:"${name}", ayahCount:${r["no_verses"]} }`;
});

const output = `// AUTO-GENERATED by scripts/generate-verse-data.ts — do not edit manually.
// Re-run if verses.csv or surahs.csv changes.

export interface VerseEntry {
  surahId: number;
  ayah: number;
  orderInQuran: number;
  reverseIndex: number;
  pageNo: number;
  lettersCount: number;
  weightOnPage: number;
  beginVerse: string;
}

export interface SurahData {
  id: number;
  name: string;
  ayahCount: number;
}

export const VERSES: VerseEntry[] = [
${verseLines.join(",\n")},
];

export const SURAHS_DATA: SurahData[] = [
${surahLines.join(",\n")},
];

export const BY_POSITION: Record<number, Record<number, VerseEntry>> = (() => {
  const m: Record<number, Record<number, VerseEntry>> = {};
  for (const v of VERSES) {
    if (!m[v.surahId]) m[v.surahId] = {};
    m[v.surahId][v.ayah] = v;
  }
  return m;
})();

export const BY_REVERSE: VerseEntry[] = [...VERSES].sort(
  (a, b) => a.reverseIndex - b.reverseIndex
);

export const REVERSE_IDX_TO_ARRAY_IDX: Record<number, number> = (() => {
  const m: Record<number, number> = {};
  BY_REVERSE.forEach((v, i) => { m[v.reverseIndex] = i; });
  return m;
})();
`;

fs.mkdirSync(path.dirname(OUT), { recursive: true });
fs.writeFileSync(OUT, output, "utf-8");
console.log(`Written ${verseRows.length} verses, ${surahRows.length} surahs → ${OUT}`);
```

```typescript
export interface WalkResult {
  from: QuranPosition;
  to: QuranPosition;
  pagesUsed: number;
}

/**
 * Walk verse-by-verse from `start`, accumulating `weightOnPage` until
 * `pageBudget` is exhausted. Matches the Python `generate_plan_by_amount`
 * algorithm exactly.
 *
 * @param start       Starting position (inclusive)
 * @param pageBudget  Maximum pages to consume (e.g., 0.467, 1.0, 5.0)
 * @param direction   "ascending" (Al-Fatihah → An-Nas) or "descending" (reverse)
 * @param stopPlace   Optional surah_id — stop before entering this surah
 */
export function walkByWeight(
  start: QuranPosition,
  pageBudget: number,
  direction: Direction,
  stopPlace?: number
): WalkResult {
  const startEntry = BY_POSITION[start.surah]?.[start.ayah];
  if (!startEntry) {
    return { from: start, to: start, pagesUsed: 0 };
  }

  let accumulated = 0;
  let lastAccepted: QuranPosition = start;
  let currentSurahId = start.surah;

  // Build the iteration sequence based on direction.
  // Ascending: iterate VERSES by orderInQuran (already sorted).
  // Descending: iterate BY_REVERSE by reverseIndex.
  const startIdx =
    direction === "ascending"
      ? startEntry.orderInQuran - 1  // VERSES is 0-indexed, orderInQuran is 1-based
      : REVERSE_IDX_TO_ARRAY_IDX[startEntry.reverseIndex];

  const arr = direction === "ascending" ? VERSES : BY_REVERSE;

  for (let i = startIdx; i < arr.length; i++) {
    const verse = arr[i];

    // Hard stop: never enter stopPlace surah
    if (stopPlace !== undefined && verse.surahId === stopPlace) break;

    // 10% surah-snap rule: if we're crossing into a new surah and remaining
    // budget < 10% of original, stop at the boundary.
    if (verse.surahId !== currentSurahId) {
      const remaining = pageBudget - accumulated;
      if (remaining < pageBudget * 0.10) break;
      currentSurahId = verse.surahId;
    }

    // Over budget?
    if (accumulated + verse.weightOnPage > pageBudget * 1.001) break;

    accumulated += verse.weightOnPage;
    lastAccepted = { surah: verse.surahId, ayah: verse.ayah };
  }

  // 110% extension rule: if finishing the current surah costs ≤ 110% of budget,
  // extend to complete it.
  const surahData = SURAHS_DATA[lastAccepted.surah - 1];
  if (surahData && lastAccepted.ayah < surahData.ayahCount) {
    let extraCost = 0;
    const extStart =
      direction === "ascending"
        ? BY_POSITION[lastAccepted.surah]?.[lastAccepted.ayah + 1]?.orderInQuran ?? null
        : null; // 110% extension only applies to ascending (completing forward)

    if (extStart !== null && direction === "ascending") {
      for (let i = extStart - 1; i < VERSES.length; i++) {
        const v = VERSES[i];
        if (v.surahId !== lastAccepted.surah) break;
        extraCost += v.weightOnPage;
      }
      if (accumulated + extraCost <= pageBudget * 1.10) {
        accumulated += extraCost;
        lastAccepted = { surah: lastAccepted.surah, ayah: surahData.ayahCount };
      }
    }
  }

  return { from: start, to: lastAccepted, pagesUsed: accumulated };
}

/**
 * Get the verse entry for a position. Returns null if not found.
 */
export function getVerseEntry(pos: QuranPosition) {
  return BY_POSITION[pos.surah]?.[pos.ayah] ?? null;
}

/**
 * Get the next verse entry in the given direction. Returns null at Quran boundary.
 */
export function getNextVerseEntry(pos: QuranPosition, direction: Direction) {
  const entry = BY_POSITION[pos.surah]?.[pos.ayah];
  if (!entry) return null;
  if (direction === "ascending") {
    return VERSES[entry.orderInQuran] ?? null; // orderInQuran is 1-based, array is 0-based
  }
  const revIdx = REVERSE_IDX_TO_ARRAY_IDX[entry.reverseIndex];
  if (revIdx === undefined) return null;
  return BY_REVERSE[revIdx + 1] ?? null;
}

/**
 * Accumulate weightOnPage from `from` to `to` (inclusive), in the given
 * direction. Used by walkWithSkips to measure block cost after migration.
 */
export function weightBetween(
  from: QuranPosition,
  to: QuranPosition,
  direction: Direction
): number {
  const fromEntry = BY_POSITION[from.surah]?.[from.ayah];
  const toEntry = BY_POSITION[to.surah]?.[to.ayah];
  if (!fromEntry || !toEntry) return 0;

  if (direction === "ascending") {
    const startIdx = fromEntry.orderInQuran - 1;
    const endIdx = toEntry.orderInQuran - 1;
    let sum = 0;
    for (let i = startIdx; i <= endIdx && i < VERSES.length; i++) {
      sum += VERSES[i].weightOnPage;
    }
    return sum;
  } else {
    const startRevPos = REVERSE_IDX_TO_ARRAY_IDX[fromEntry.reverseIndex];
    const endRevPos = REVERSE_IDX_TO_ARRAY_IDX[toEntry.reverseIndex];
    if (startRevPos === undefined || endRevPos === undefined) return 0;
    const lo = Math.min(startRevPos, endRevPos);
    const hi = Math.max(startRevPos, endRevPos);
    let sum = 0;
    for (let i = lo; i <= hi && i < BY_REVERSE.length; i++) {
      sum += BY_REVERSE[i].weightOnPage;
    }
    return sum;
  }
}
```

```typescript
import type { QuranPosition, Direction, PositionRange } from "@/lib/quran/types";
import { LINES_PER_PAGE } from "@/lib/quran/constants";
import { walkByWeight, getNextVerseEntry, normalizeRange } from "./helpers";
import { BY_POSITION } from "@/lib/quran/verseData";

export interface NewMemorizationResult {
  from: QuranPosition;
  to: QuranPosition;
  range: PositionRange;
  newCursor: QuranPosition;
  /** The chronologically last ayah memorized in this session. */
  frontier: QuranPosition;
}

/**
 * Calculate the new memorization section for one assignment.
 *
 * The student always reads forward (ayah 1 → last) within each surah.
 * The `direction` determines which surah they go to next:
 *   - ascending: surah N → surah N+1
 *   - descending: surah N → surah N-1
 *
 * Uses walkByWeight to match the Python generate_plan_by_amount algorithm.
 */
export function calculateNewMemorization(
  cursor: QuranPosition,
  linesPerSession: number,
  direction: Direction
): NewMemorizationResult | null {
  if (linesPerSession <= 0) return null;

  const pageBudget = linesPerSession / LINES_PER_PAGE;
  const walked = walkByWeight(cursor, pageBudget, direction);

  const from = walked.from;
  const to = walked.to;

  // newCursor: the next position to start memorizing
  const nextEntry = getNextVerseEntry(to, direction);
  const newCursor: QuranPosition = nextEntry
    ? { surah: nextEntry.surahId, ayah: nextEntry.ayah }
    : to;

  // range is always normalized (start <= end in Quran order)
  const range = normalizeRange(from, to);

  return { from, to, range, newCursor, frontier: to };
}
```

```typescript
/**
 * Calculate the minor revision zone: the last `minorRevPages` pages of
 * memorized content, ending at the frontier (last ayah memorized before
 * today's new task).
 *
 * Walks BACKWARD from the frontier (in the opposite of the review direction,
 * which is the memorization direction itself) accumulating weight_on_page.
 *
 * For descending memorization (ascending review): walking backward from
 * frontier = walking descending (toward lower surah numbers = older material).
 *
 * For ascending memorization (descending review): walking backward from
 * frontier = walking ascending is wrong — we walk descending (toward lower
 * surah numbers = older material).
 *
 * Wait — "backward through memorization time" means toward older material:
 * - descending mem: older material is at HIGHER surah numbers. Walk ascending.
 * - ascending mem: older material is at LOWER surah numbers. Walk descending.
 *
 * @param frontier     Last ayah memorized before today (null on assignment 1)
 * @param minorRevPages Budget in pages
 * @param memStart     Oldest memorization boundary (clamp — don't go past)
 * @param memDirection Student's memorization direction
 */
export function calculateMinorRevision(
  frontier: QuranPosition | null,
  minorRevPages: number,
  memStart: QuranPosition,
  memDirection: Direction
): MinorRevisionResult | null {
  if (!frontier || minorRevPages <= 0) return null;

  // Walking "backward through memorization time" means toward older material:
  // - descending mem (newer = lower surah): older = higher surah → walk ascending
  // - ascending mem (newer = higher surah): older = lower surah → walk descending
  const walkDir: Direction = memDirection === "descending" ? "ascending" : "descending";

  // stopPlace: the surah of memStart — don't walk past the oldest boundary
  const stopPlace = memStart.surah;

  const walked = walkByWeight(frontier, minorRevPages, walkDir, stopPlace);

  // `from` is the older end (walked.to in the backward direction)
  // `to` is the frontier (the newer end)
  const olderEnd = walked.to;
  const newerEnd = frontier;

  return {
    from: olderEnd,
    to: newerEnd,
    range: normalizeRange(olderEnd, newerEnd),
  };
}
```
